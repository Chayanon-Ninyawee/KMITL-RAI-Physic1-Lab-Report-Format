\begin{itemize}
	\item Calculate the force acting on a 10 kg mass accelerating at 2 m/s\textsuperscript{2}.  % NOTE: EXAMPLE QUESTION

	      \begin{align*}
		      F & = ma           \\
		        & = 10 \times 2  \\
		        & = 20 \text{ N}
	      \end{align*}

	\item Determine the work done by a force of 15 N moving an object 3 m in the direction of the force.  % NOTE: EXAMPLE QUESTION

	      \begin{align*}
		      W & = F \times d   \\
		        & = 15 \times 3  \\
		        & = 45 \text{ J}
	      \end{align*}

	\item Find the kinetic energy of an object with mass 5 kg moving at 4 m/s.  % NOTE: EXAMPLE QUESTION

	      \begin{align*}
		      KE & = \frac{1}{2} m v^2               \\
		         & = \frac{1}{2} \times 5 \times 4^2 \\
		         & = 40 \text{ J}
	      \end{align*}
\end{itemize}
